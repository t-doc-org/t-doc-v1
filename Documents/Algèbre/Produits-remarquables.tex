\section{Théorie}
Les trois équations suivantes sont appelées \textbf{produits remarquables}.
$$\begin{array}{rcl}
(a+b)^2 &=& a^2+2ab+b^2 \\
(a-b)^2 &=& a^2-2ab+b^2 \\
(a+b)(a-b) &=& a^2-b^2
\end{array}$$

\section{Exercice}
Calculer à l'aide des produits remarquables.
\begin{enumerate}
\item $(x+3)^2=$
\item $(4-3x)^2=$
\item $(2x+3)(2x-3)=$
\end{enumerate}

\begin{solution}
\begin{enumerate}
\item $x^2+6x+9$
\item $16-24x+9x^2$
\item $4x^2-9$
\end{enumerate}

\section{Exercice}
Calculer à l'aide des produits remarquables.
\begin{multicols}{2}
\begin{enumerate}
\item $(x-y)^2=$
\item $(x-1)^2=$
\item $(2c+1)(2c-1)=$
\item $(x-2)(x+2)=$
\item $(2x+y)^2=$
\item $(qr-r)^2=$
\item $(ef^2-f^3)^2=$
\item $(z^3-1)(z^3+1)=$
\item $(x^3+x^2y)^2=$
\item $(5y+z)(5y-z)=$
\end{enumerate}
\end{multicols}

\begin{solution}
\begin{enumerate}
\item $x^2-2xy+y^2$
\item $x^2-2x+1$
\item $4c^2-1$
\item $x^2-4$
\item $4x^2+4xy+y^2$
\item $q^2r^2-2qr^2+r^2$
\item $e^2f^4-2ef^5+f^6$
\item $z^6-1$
\item $x^6+2x^5y+x^4y^2$
\item $25y^2-z^2$
\end{enumerate}
\end{solution}
